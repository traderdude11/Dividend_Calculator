<script>

let totalInvestment = 0;

let totalProfit = 0;


function addTrade() {

  let date = document.getElementById("exDate").value;

  let company = document.getElementById("company").value;

  let investment = parseFloat(document.getElementById("investment").value);

  let dividend = parseFloat(document.getElementById("dividend").value);

  let buyPrice = parseFloat(document.getElementById("buyPrice").value);

  let sellPrice = parseFloat(document.getElementById("sellPrice").value);


  if (!investment || !buyPrice) return alert("Enter valid inputs");


  let shares = investment / buyPrice;

  let payout = shares * dividend;

  let tradingPL = shares * (sellPrice - buyPrice);

  let totalPL = payout + tradingPL;

  let tax = totalPL > 0 ? totalPL * 0.2 : 0;

  let finalPL = totalPL - tax;


  let table = document.getElementById("table");


  let row = table.insertRow();

  row.innerHTML = `

    <td>${date}</td>

    <td>${company}</td>

    <td>${shares.toFixed(0)}</td>

    <td>₹${payout.toFixed(0)}</td>

    <td>₹${tradingPL.toFixed(0)}</td>

    <td>₹${totalPL.toFixed(0)}</td>

    <td>₹${tax.toFixed(0)}</td>

    <td class="${finalPL>=0?'profit':'loss'}">₹${finalPL.toFixed(0)}</td>

  `;


  totalInvestment += investment;

  totalProfit += finalPL;


  document.getElementById("totalInv").innerText = totalInvestment.toFixed(0);

  document.getElementById("totalProfit").innerText = totalProfit.toFixed(0);

}

</script>
